rules_version = '2';

service cloud.firestore {
  match /databases/{database}/documents {
    
    // Helper function to check if user is authenticated
    function isAuthenticated() {
      return request.auth != null;
    }
    
    // Helper function to check if user owns the document
    function isOwner(userId) {
      return isAuthenticated() && request.auth.uid == userId;
    }
    
    // Helper function to check if user is admin
    function isAdmin() {
      return isAuthenticated() && 
        exists(/databases/$(database)/documents/adminUsers/$(request.auth.uid));
    }
    
    // Alternative admin check using users collection role field
    function isAdminByRole() {
      return isAuthenticated() && 
        get(/databases/$(database)/documents/users/$(request.auth.uid)).data.role == 'admin';
    }
    
    // Combined admin check (checks both adminUsers collection and users.role)
    function isAdminUser() {
      return isAdmin() || isAdminByRole();
    }
    
    // Users collection - users can read/write their own data
    match /users/{userId} {
      allow read: if isOwner(userId) || isAdminUser();
      allow create: if isAuthenticated() && request.auth.uid == userId;
      allow update: if isOwner(userId);
      allow delete: if isAdminUser();
    }
    
    // Carts collection - users can only access their own cart
    match /carts/{cartId} {
      allow read: if isAuthenticated();
      allow write: if isAuthenticated();
      allow create: if isAuthenticated();
      
      // Cart items subcollection
      match /items/{itemId} {
        allow read, write: if isAuthenticated();
      }
    }
    
    // Wishlists collection - users can only access their own wishlist
    match /wishlists/{wishlistId} {
      allow read: if isAuthenticated();
      allow write: if isAuthenticated();
      allow create: if isAuthenticated();
      
      // Wishlist items subcollection
      match /items/{itemId} {
        allow read, write: if isAuthenticated();
      }
    }
    
    // Products collection - public read, admin write
    match /products/{productId} {
      allow read: if true;
      allow create, update, delete: if isAdminUser();
    }
    
    // Collections - public read, admin write
    match /collections/{collectionId} {
      allow read: if true;
      allow create, update, delete: if isAdminUser();
    }
    
    // Featured Products - public read, admin write
    match /featuredProducts/{featuredId} {
      allow read: if true;
      allow create, update, delete: if isAdminUser();
    }
    
    // Orders collection - IMPORTANT: Allow list queries for authenticated users
    match /orders/{orderId} {
      // Allow read if user owns the order OR is admin
      // For list queries, Firebase checks after fetching, so we allow authenticated reads
      allow read: if isAuthenticated();
      allow create: if isAuthenticated();
      allow update, delete: if isAdminUser();
    }
    
    // Reviews collection - public read, authenticated users can create
    match /reviews/{reviewId} {
      allow read: if true;
      allow create: if isAuthenticated();
      allow update: if isAuthenticated() && 
        (resource.data.userId == request.auth.uid || isAdminUser());
      allow delete: if isAdminUser();
    }
    
    // Contacts collection - anyone can create, only admins can read/manage
    match /contacts/{contactId} {
      allow create: if true;
      allow read, update, delete: if isAdminUser();
    }
    
    // Admin users collection - authenticated users can read, only admins can write
    match /adminUsers/{userId} {
      allow read: if isAuthenticated();
      allow write: if isAdminUser();
    }
    
    // Payment Transactions - only admins can access
    match /paymentTransactions/{transactionId} {
      allow read, write: if isAdminUser();
    }
    
    // Inventory Logs - only admins can access
    match /inventoryLogs/{logId} {
      allow read, write: if isAdminUser();
    }
    
    // Feedback - public read, authenticated create, admin manage
    match /feedback/{feedbackId} {
      allow read: if true;
      allow create: if isAuthenticated();
      allow update, delete: if isAdminUser();
    }
    
    // Support Tickets - users can manage their own, admins can see all
    match /supportTickets/{ticketId} {
      allow read: if isAuthenticated() && 
        (resource.data.userId == request.auth.uid || isAdminUser());
      allow create: if isAuthenticated();
      allow update: if isAuthenticated() && 
        (resource.data.userId == request.auth.uid || isAdminUser());
      allow delete: if isAdminUser();
    }
  }
}
